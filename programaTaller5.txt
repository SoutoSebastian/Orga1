.data
tablaCalificaciones: .half 5523
    .byte 3
    .half 8754
    .byte 6
    .half 991
    .byte 8
    .half 867
    .byte 4
    .half 0

.text

.global main
main:
    la a0 tablaCalificaciones
    call sumaImparNotas
halt:
    j halt

.global sumaImparNotas
sumaImparNotas:
    li t0, 0 #acumulador
    mv t1, a0
    
    ciclo:
        lhu t2, 0(t1)
        beq t2, zero, fin
        
        andi t3, t2, 1
        beq t3, zero, siguiente
        
        lbu t4, 2(t1)
        add t0, t0, t4
        
        
    siguiente:
        addi t1, t1, 3
        j ciclo    
    
    fin:
        mv a0, t0
        ret
